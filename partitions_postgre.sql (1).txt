CREATE TABLE IF NOT EXISTS public.measurement
(
    city_id integer NOT NULL,
    logdate date NOT NULL,
    peaktemp integer,
    unitsales integer
) PARTITION BY RANGE (logdate);

ALTER TABLE IF EXISTS public.measurement
    OWNER to postgres;

-- Partitions SQL

CREATE TABLE public.measurement_y2006m02 PARTITION OF public.measurement
    FOR VALUES FROM ('2006-02-01') TO ('2006-03-01');

ALTER TABLE IF EXISTS public.measurement_y2006m02
    OWNER to postgres;

PARTITION BY LIST (region);
